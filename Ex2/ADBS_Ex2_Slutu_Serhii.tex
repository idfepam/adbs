\section*{Exercise 1 (Distributed Joins) [3 points]}
% ============================================================

\subsection*{Setup}

We have a distributed database with 4 sites and three relations:

\begin{center}
\begin{tabular}{llrr}
\toprule
\textbf{Relation} & \textbf{Site} & \textbf{\# Records} & \textbf{Record Size (bytes)} \\
\midrule
Authors    & 1 & 4\,000  & 450 \\
Books      & 2 & 50\,000 & 400 \\
Libraries  & 3 & 10\,000 & 750 \\
\bottomrule
\end{tabular}
\end{center}

Relevant attribute sizes (bytes):

\begin{center}
\begin{tabular}{lll}
\toprule
\textbf{Relation} & \textbf{Attribute} & \textbf{Size} \\
\midrule
Authors   & authid  & 12 \\
Authors   & name    & 58 \\
Books     & author  & 10 \\
Books     & title   & 30 \\
Books     & library & 10 \\
Libraries & libid   & 10 \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Query 1:}
\[
  \pi_{\text{name},\,\text{title}}\!\bigl(\Authors \join_{\text{authid}=\text{author}} \Books\bigr)
\]
Result desired at \textbf{Site 4}. Join selectivity: $\tfrac{1}{9500}$.

% -----------------------------------------------------------
\subsection*{Part (a) — Communication costs without early projection}
% -----------------------------------------------------------

\textbf{Assumptions and cardinality estimates:}

\begin{itemize}[nosep]
  \item All records and attributes are fixed size.
  \item Join result size: $|\Authors \join \Books| = |\Authors| \times |\Books| \times \tfrac{1}{9500}
        = 4000 \times 50000 / 9500 \approx 21{,}053$ rows.
  \item Result record size (name + title): $58 + 30 = 88$ bytes.
        Total result size: $21{,}053 \times 88 = 1{,}852{,}664$ bytes.
  \item Semi-join estimate: $|A \lsj B| = \min\{|A \join B|,\, |A|\}$.
  \item Communication cost = bytes transferred over the network (local computation is free).
\end{itemize}

\medskip

\textbf{Strategy 1 — Send both tables to Site 4, join there:}
\[
  \underbrace{4000 \times 450}_{\Authors \to \text{S4}} +
  \underbrace{50000 \times 400}_{\Books \to \text{S4}}
  = 1{,}800{,}000 + 20{,}000{,}000 = \mathbf{21{,}800{,}000\ \text{bytes}}
\]

\medskip

\textbf{Strategy 2 — Send \Authors\ to Site 2, join there, send result to Site 4:}
\[
  \underbrace{4000 \times 450}_{\Authors \to \text{S2}} +
  \underbrace{21{,}053 \times 88}_{\text{result} \to \text{S4}}
  = 1{,}800{,}000 + 1{,}852{,}664 = \mathbf{3{,}652{,}664\ \text{bytes}}
\]

\medskip

\textbf{Strategy 3 — Send \Books\ to Site 1, join there, send result to Site 4:}
\[
  \underbrace{50000 \times 400}_{\Books \to \text{S1}} +
  \underbrace{21{,}053 \times 88}_{\text{result} \to \text{S4}}
  = 20{,}000{,}000 + 1{,}852{,}664 = \mathbf{21{,}852{,}664\ \text{bytes}}
\]

\medskip

\textbf{Strategy 4 — Semi-join: send \Authors.authid\ to Site 2, filter \Books, ship back to Site 1, join, result to Site 4:}

\begin{enumerate}[nosep,label=\arabic*.]
  \item Send $\pi_{\text{authid}}(\Authors)$ from S1 to S2: $4000 \times 12 = 48{,}000$ bytes.
  \item $\Books \lsj \Authors$: $\min(21{,}053,\, 50{,}000) = 21{,}053$ Books rows survive.
        Send back to S1: $21{,}053 \times 400 = 8{,}421{,}200$ bytes.
  \item Full join on S1; send result to S4: $21{,}053 \times 88 = 1{,}852{,}664$ bytes.
\end{enumerate}
\[
  48{,}000 + 8{,}421{,}200 + 1{,}852{,}664 = \mathbf{10{,}321{,}864\ \text{bytes}}
\]

\medskip

\textbf{Strategy 5 — Semi-join in opposite direction: send \Books.author\ to Site 1, filter \Authors, ship back to Site 2, join, result to Site 4:}

\begin{enumerate}[nosep,label=\arabic*.]
  \item Send $\pi_{\text{author}}(\Books)$ from S2 to S1: $50{,}000 \times 10 = 500{,}000$ bytes.
  \item $\Authors \lsj \Books$: $\min(21{,}053,\, 4{,}000) = 4{,}000$ Authors rows survive
        (the join result exceeds $|\Authors|$, so all Authors are retained).
        Send back to S2: $4{,}000 \times 450 = 1{,}800{,}000$ bytes.
  \item Full join on S2; send result to S4: $21{,}053 \times 88 = 1{,}852{,}664$ bytes.
\end{enumerate}
\[
  500{,}000 + 1{,}800{,}000 + 1{,}852{,}664 = \mathbf{4{,}152{,}664\ \text{bytes}}
\]

\medskip

\textbf{Summary for part (a):}

\begin{center}
\begin{tabular}{lrr}
\toprule
\textbf{Strategy} & \textbf{Cost (bytes)} & \textbf{Cost (MB)} \\
\midrule
1 — Both to S4              & 21{,}800{,}000  & $\approx 20.79$ \\
2 — Authors to S2 \checkmark & 3{,}652{,}664  & $\approx 3.48$ \\
3 — Books to S1             & 21{,}852{,}664  & $\approx 20.84$ \\
4 — Semi-join A$\to$B       & 10{,}321{,}864  & $\approx 9.84$ \\
5 — Semi-join B$\to$A       &  4{,}152{,}664  & $\approx 3.96$ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Minimum: Strategy 2} at $\approx 3{,}652{,}664$ bytes.

Note: the semi-join strategies do not help here because the join selectivity is not low enough. The join result ($\approx 21{,}053$ rows) is larger than $|\Authors|$ ($4{,}000$), so Strategy 5 retains all Authors anyway. Strategy 4 ships $42\%$ of \Books\ back, which is far more expensive than simply sending the small \Authors\ table to \Books\ (Strategy 2).

% -----------------------------------------------------------
\subsection*{Part (a*) — Communication costs with early projection}
% -----------------------------------------------------------

Since Query 1 only needs \texttt{name} and \texttt{title} in the result, we project each relation to the minimum columns needed before any transfer:

\begin{itemize}[nosep]
  \item $\pi_{\text{authid},\,\text{name}}(\Authors)$: $12 + 58 = 70$ bytes/row, total $4000 \times 70 = 280{,}000$ bytes.
  \item $\pi_{\text{author},\,\text{title}}(\Books)$: $10 + 30 = 40$ bytes/row, total $50{,}000 \times 40 = 2{,}000{,}000$ bytes.
  \item Result record: $58 + 30 = 88$ bytes (unchanged). $21{,}053 \times 88 = 1{,}852{,}664$ bytes.
\end{itemize}

\medskip

\textbf{Strategy 1*:}
\[
  280{,}000 + 2{,}000{,}000 = \mathbf{2{,}280{,}000\ \text{bytes}}
\]

\textbf{Strategy 2*:}
\[
  280{,}000 + 1{,}852{,}664 = \mathbf{2{,}132{,}664\ \text{bytes}}
\]

\textbf{Strategy 3*:}
\[
  2{,}000{,}000 + 1{,}852{,}664 = \mathbf{3{,}852{,}664\ \text{bytes}}
\]

\textbf{Strategy 4* (semi-join, project Books before sending back):}
\begin{enumerate}[nosep,label=\arabic*.]
  \item $\pi_{\text{authid}}(\Authors) \to \text{S2}$: $48{,}000$ bytes.
  \item $\pi_{\text{author},\text{title}}(\Books \lsj \Authors) \to \text{S1}$: $21{,}053 \times 40 = 842{,}120$ bytes.
  \item Result $\to$ S4: $1{,}852{,}664$ bytes.
\end{enumerate}
\[
  48{,}000 + 842{,}120 + 1{,}852{,}664 = \mathbf{2{,}742{,}784\ \text{bytes}}
\]

\textbf{Strategy 5* (semi-join, project Authors before sending back):}
\begin{enumerate}[nosep,label=\arabic*.]
  \item $\pi_{\text{author}}(\Books) \to \text{S1}$: $500{,}000$ bytes.
  \item $\pi_{\text{authid},\text{name}}(\Authors \lsj \Books) \to \text{S2}$: $4{,}000 \times 70 = 280{,}000$ bytes.
  \item Result $\to$ S4: $1{,}852{,}664$ bytes.
\end{enumerate}
\[
  500{,}000 + 280{,}000 + 1{,}852{,}664 = \mathbf{2{,}632{,}664\ \text{bytes}}
\]

\medskip

\textbf{Summary for part (a*):}

\begin{center}
\begin{tabular}{lrr}
\toprule
\textbf{Strategy} & \textbf{Cost (bytes)} & \textbf{Cost (MB)} \\
\midrule
1* — Both to S4              & 2{,}280{,}000  & $\approx 2.17$ \\
2* — Authors to S2 \checkmark & 2{,}132{,}664 & $\approx 2.03$ \\
3* — Books to S1             & 3{,}852{,}664  & $\approx 3.67$ \\
4* — Semi-join A$\to$B       & 2{,}742{,}784  & $\approx 2.61$ \\
5* — Semi-join B$\to$A       & 2{,}632{,}664  & $\approx 2.51$ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Minimum: Strategy 2*} at $\approx 2{,}132{,}664$ bytes.

Early projection significantly reduces all costs. The most dramatic improvement is in strategies that transfer \Books\ (from 20 MB down to 2 MB for the full table, and from 8.4 MB down to 842 KB for the semi-join result in Strategy 4). Strategy 2* still wins: sending the small projected \Authors\ to Site 2 (280 KB) is cheaper than any alternative.

% -----------------------------------------------------------
\subsection*{Part (b) — Three-way join with Libraries}
% -----------------------------------------------------------

\textbf{Query 2:}
\[
  \Libraries \join_{\text{libid}=\text{library}}
    \pi_{\text{name},\,\text{library}}\!\bigl(\Authors \join_{\text{authid}=\text{author}} \Books\bigr)
\]
Join selectivity for the outer join (\Libraries\ with inner result): $\tfrac{1}{4000}$.

\medskip

\textbf{Inner join projection for Query 2:}

The inner join now needs \texttt{library} from \Books\ (to join with \Libraries) and \texttt{name} from \Authors. So:
\begin{itemize}[nosep]
  \item $\pi_{\text{authid},\,\text{name}}(\Authors)$: $70$ bytes/row (unchanged).
  \item $\pi_{\text{author},\,\text{library}}(\Books)$: $10 + 10 = 20$ bytes/row,
        total $50{,}000 \times 20 = 1{,}000{,}000$ bytes.
  \item Inner result: $\pi_{\text{name},\,\text{library}}(\Authors \join \Books)$
        $= 21{,}053$ rows $\times$ $(58 + 10) = 68$ bytes $= 1{,}431{,}604$ bytes.
  \item Outer result: $21{,}053 \times 10{,}000 \times \tfrac{1}{4000} \approx 52{,}633$ rows (small).
\end{itemize}

\medskip

\textbf{Key observation for skipping strategies:}

The full \Libraries\ table costs $10{,}000 \times 750 = 7{,}500{,}000$ bytes to transfer. Any strategy that ships \Libraries\ to another site will incur at least 7.5 MB just for that step. Since Strategy 1' below costs only $\approx 1{,}280{,}000$ bytes total, all strategies involving moving \Libraries\ can be immediately eliminated.

\medskip

Since \Libraries\ is at Site 3, the optimal approach is to bring the inner join result to Site 3. The total cost for any strategy is:
\[
  \text{inner join cost} + \text{cost to ship inner result to S3} + \text{small (result to S4)}
\]

The inner result size is fixed at $1{,}431{,}604$ bytes regardless of where it is computed. The exception is if we ship both \Authors\ and \Books\ directly to Site 3, avoiding a separate "inner result transfer" step entirely.

\medskip

\textbf{Strategy 1' — Send \Authors\ and \Books\ directly to Site 3, join everything there:}
\[
  \underbrace{4000 \times 70}_{\Authors \to \text{S3}}
  + \underbrace{50{,}000 \times 20}_{\Books \to \text{S3}}
  = 280{,}000 + 1{,}000{,}000 = \mathbf{1{,}280{,}000\ \text{bytes}}
\]
(Both inner join and outer join are computed locally on Site 3. Result goes to Site 4: negligible.)

\medskip

\textbf{Strategy 2' — Send \Authors\ to Site 2, compute inner join there, send result to Site 3:}
\[
  \underbrace{280{,}000}_{\Authors \to \text{S2}}
  + \underbrace{1{,}431{,}604}_{\text{inner result} \to \text{S3}}
  = \mathbf{1{,}711{,}604\ \text{bytes}}
\]

\textbf{Strategy 4' — Semi-join (as in 4*), compute inner at S1, send result to Site 3:}
\[
  48{,}000 + 21{,}053 \times 20 + 1{,}431{,}604
  = 48{,}000 + 421{,}060 + 1{,}431{,}604 = \mathbf{1{,}900{,}664\ \text{bytes}}
\]

Strategies 3', 5' are skipped: Strategy 3' requires sending \Books\ (1 MB) to S1 plus the inner result (1.43 MB) to S3, totalling $\approx 2.43$ MB — worse than Strategy 1'. Strategy 5' requires the \Books\ key transfer (500 KB) which alone is larger than the advantage over Strategy 1'.

\medskip

\textbf{Summary for part (b):}

\begin{center}
\begin{tabular}{lrr}
\toprule
\textbf{Strategy} & \textbf{Cost (bytes)} & \textbf{Cost (MB)} \\
\midrule
1' — Both to S3 \checkmark & 1{,}280{,}000  & $\approx 1.22$ \\
2' — Authors to S2, result to S3 & 1{,}711{,}604 & $\approx 1.63$ \\
4' — Semi-join, result to S3    & 1{,}900{,}664 & $\approx 1.81$ \\
3', 5' (skipped) & $\geq 2{,}430{,}664$ & $\geq 2.32$ \\
Any strategy moving Libraries & $\geq 7{,}500{,}000$ & $\geq 7.15$ \\
\bottomrule
\end{tabular}
\end{center}

\textbf{Minimum: Strategy 1'} — send both projected \Authors\ and \Books\ directly to Site 3, perform all joins locally there, then ship the small final result to Site 4. Total cost: $\approx 1{,}280{,}000$ bytes.

The key insight: for three-way joins it can be cheaper to ship all smaller tables to the site of the largest table (Site 3 for \Libraries) rather than first computing an intermediate result and then transferring it.

% -----------------------------------------------------------
\subsection*{Part (b) — Cost analysis per stage}
% -----------------------------------------------------------

We express costs in terms of graph properties. Let $|U|$ = number of users, $|V|$ = number of items, $|E|$ = number of edges (user-item interactions). $\deg(v)$ = degree of item $v$ (number of users connected to it). $CI(u_1,u_2)$ = set of common items of users $u_1, u_2$. $S(u)$ = set of users sharing at least one item with $u$.

\medskip

\textbf{Stage 1 (Mapper 1 / Reducer 1):}
\begin{itemize}[nosep]
  \item \textbf{Communication cost}: $|E|$ — each edge $(u,v)$ is emitted exactly once as $(v,u)$. The identity reducer passes all pairs through unchanged.
  \item \textbf{Replication rate}: $1$ — each edge produces exactly one key-value pair in the mapper output.
  \item \textbf{Reducer size}: $\deg(v)$ for the reducer handling item $v$ (it receives one value per user connected to $v$). Maximum reducer size = $\max_{v \in V} \deg(v)$.
\end{itemize}

\medskip

\textbf{Stage 2 (Mapper 2 / Reducer 2):}
\begin{itemize}[nosep]
  \item \textbf{Communication cost}: $\displaystyle\sum_{v \in V} \binom{\deg(v)}{2} = \sum_{v \in V} \frac{\deg(v)(\deg(v)-1)}{2}$ — for each item $v$, all user pairs that share $v$ are emitted.
  \item \textbf{Replication rate}: $\displaystyle\frac{\sum_{v \in V} \binom{\deg(v)}{2}}{|E|}$ — mapper input size is $|E|$ (output of Stage 1), mapper output is $\sum_v \binom{\deg(v)}{2}$ pairs.
  \item \textbf{Reducer size}: $|CI(u_1, u_2)|$ for the reducer handling pair $(u_1, u_2)$ — it receives one 1 per common item of $u_1$ and $u_2$. Maximum reducer size = $\max_{u_1,u_2} |CI(u_1,u_2)|$.
\end{itemize}

\medskip

\textbf{Stage 3 (Mapper 3 / Reducer 3):}
\begin{itemize}[nosep]
  \item Let $P$ = number of unique user pairs with at least one common item (number of edges in the similarity graph before top-$k$ pruning).
  \item \textbf{Communication cost}: $2P$ — each pair $(u_1, u_2, s)$ is emitted twice: once keyed by $u_1$ and once by $u_2$. Equivalently, $P = |\{(u_1,u_2) : CI(u_1,u_2) \neq \emptyset\}|$.
  \item \textbf{Replication rate}: $2$ — each pair from Reducer 2 generates exactly 2 mapper outputs.
  \item \textbf{Reducer size}: $|S(u)|$ for the reducer handling user $u$ — it receives one candidate per user sharing at least one item with $u$. Maximum reducer size = $\max_{u \in U} |S(u)|$.
\end{itemize}

\section*{Exercise 3 (MapReduce Algorithms) [2 points]}
% ============================================================

\subsection*{Part (a) — MapReduce for $F_1(X, Y, Z)$}

\[
  F_1(X, Y, Z) = \gamma_{b}^{SUM(a)}\!\bigl(\sigma_{a > b}(X) \;-\; \pi_{a,b}(Y \lsj Z)\bigr)
\]

Relations: $X(a,b)$, $Y(a,b,c)$, $Z(b,c,d)$. Set semantics (no duplicates).

The computation requires three logical steps:
\begin{enumerate}[nosep,label=\arabic*.]
  \item Compute the semi-join $Y \lsj Z$ (retain tuples of $Y$ that have a matching $(b,c)$ in $Z$).
  \item Compute the set difference $\sigma_{a>b}(X) - \pi_{a,b}(Y \lsj Z)$.
  \item Group by $b$ and sum $a$.
\end{enumerate}

\medskip

\textbf{Round 1 — Compute $\pi_{a,b}(Y \lsj Z)$ and prepare $\sigma_{a>b}(X)$:}

\textit{Mapper:}
\begin{itemize}[nosep]
  \item For each tuple $(a, b, c) \in Y$: emit key $(b, c)$, value $(\texttt{Y}, a, b)$.
  \item For each tuple $(b, c, d) \in Z$: emit key $(b, c)$, value $(\texttt{Z})$.
  \item For each tuple $(a, b) \in X$ with $a > b$: emit key $(\texttt{X}, a, b)$, value $(\texttt{X}, a, b)$.
\end{itemize}

\textit{Reducer (for keys of the form $(b,c)$):}
\begin{itemize}[nosep]
  \item If the value list contains at least one \texttt{Z} entry, emit $(\texttt{YsZ}, a, b)$ for each \texttt{Y} entry $(a, b)$ in the list (these are tuples from $\pi_{a,b}(Y \lsj Z)$).
  \item Keys of the form $(\texttt{X}, a, b)$ are passed through as-is (identity for $X$ tuples).
\end{itemize}

\medskip

\textbf{Round 2 — Compute set difference and group-by aggregation:}

\textit{Mapper:}
\begin{itemize}[nosep]
  \item For each \texttt{X}-tuple $(a, b)$ from Round 1: emit key $(a, b)$, value $(\texttt{X})$.
  \item For each \texttt{YsZ}-tuple $(a, b)$ from Round 1: emit key $(a, b)$, value $(\texttt{YsZ})$.
\end{itemize}

\textit{Reducer:}
\begin{itemize}[nosep]
  \item For each key $(a, b)$: if the value list contains \texttt{X} but no \texttt{YsZ}, emit $(b,\, a)$ — this tuple is in the set difference. If both tags are present, the tuple is excluded (set difference removes it).
\end{itemize}

\medskip

\textbf{Round 3 — Group by $b$ and compute $SUM(a)$:}

\textit{Mapper:} For each surviving tuple $(b, a)$ from Round 2: emit key $b$, value $a$.

\textit{Reducer:} For each key $b$: emit $(b,\; \sum_{\text{values}} a)$.

\subsection*{Part (b) — MapReduce for min-max row normalization}
% -----------------------------------------------------------

Given matrix $M$ with $I$ rows and $J$ columns:
\[
  M'_{ij} = \frac{M_{ij} - \min(M_i)}{\max(M_i) - \min(M_i)}, \quad
  \text{with } M'_{ij} = 0 \text{ if } \max(M_i) = \min(M_i).
\]

\medskip

\textbf{Round 1 — Compute row min and max:}

\textit{Mapper:} For each entry $M_{ij}$: emit key $i$, value $(j, M_{ij})$.

\textit{Reducer:} For each row $i$, collect all values $\{(j, M_{ij})\}$.
\begin{itemize}[nosep]
  \item Compute $\min_i = \min_j M_{ij}$ and $\max_i = \max_j M_{ij}$.
  \item Emit $(i,\, \min_i,\, \max_i)$.
\end{itemize}

\medskip

\textbf{Round 2 — Normalize each entry:}

\textit{Mapper (input: both $M$ and the output of Round 1):}
\begin{itemize}[nosep]
  \item For each entry $M_{ij}$: emit key $i$, value $(\texttt{entry},\, j,\, M_{ij})$.
  \item For each $(i, \min_i, \max_i)$ from Round 1: emit key $i$, value $(\texttt{stats},\, \min_i,\, \max_i)$.
\end{itemize}

\textit{Reducer:} For each row $i$, retrieve $\min_i$ and $\max_i$ from the \texttt{stats} entry, then for each \texttt{entry} $(j, M_{ij})$:
\begin{itemize}[nosep]
  \item If $\max_i = \min_i$: emit $(i,\, j,\, 0)$.
  \item Otherwise: emit $\left(i,\, j,\;\dfrac{M_{ij} - \min_i}{\max_i - \min_i}\right)$.
\end{itemize}
